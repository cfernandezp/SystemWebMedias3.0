  String? _validateEmail(String? value) {
    if (value == null || value.isEmpty) return 'Email es requerido';
    final emailRegex = RegExp(r'^[^\s@]+@[^\s@]+\.[^\s@]+$');
    if (!emailRegex.hasMatch(value)) return 'Formato de email inv√°lido';
    return null;
  }

  String? _validatePassword(String? value) {
    if (value == null || value.isEmpty) return 'Contrase√±a es requerida';
    if (value.length < 8) return 'Contrase√±a debe tener al menos 8 caracteres';
    return null;
  }

  String? _validateConfirmPassword(String? value) {
    if (value == null || value.isEmpty) return 'Confirmar contrase√±a es requerido';
    if (value != _passwordController.text) return 'Las contrase√±as no coinciden';
    return null;
  }

  String? _validateNombreCompleto(String? value) {
    if (value == null || value.trim().isEmpty) return 'Nombre completo es requerido';
    return null;
  }

  void _handleRegister() {
    if (_formKey.currentState!.validate()) {
      final request = RegisterRequestModel(
        email: _emailController.text,
        password: _passwordController.text,
        confirmPassword: _confirmPasswordController.text,
        nombreCompleto: _nombreCompletoController.text,
      );

      context.read<AuthBloc>().add(AuthRegisterRequested(request));
    }
  }
}
